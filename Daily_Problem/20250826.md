```cpp
using i32 = int;
using i64 = long long;
using pii = pair<int, int>;
using pll = pair<i64, i64>;
const int mod = 1000000007, mod0 = 998244353;
class Solution {
public:
    bool isValidSudoku(vector<vector<char>>& board) {
        int n = board.size();
        for (int i = 0; i < n; ++i) {
            vector<bool> row(10), col(10);
            for (int j = 0; j < n; ++j) {
                if (board[i][j] != '.') {
                    int v = board[i][j] - '0';
                    if (row[v])
                        return false;
                    row[v] = true;
                }

                if (board[j][i] != '.') {
                    int v0 = board[j][i] - '0';
                    if (col[v0])
                        return false;
                    col[v0] = true;
                }
            }
        }

        for (int i = 0; i < n; i += 3) {
            for (int j = 0; j < n; j += 3) {
                vector<bool> vis(10);
                for (int p = 0; p < 3; ++p) {
                    for (int q = 0; q < 3; ++q) {
                        if (board[i + p][j + q] == '.')
                            continue;

                        int v = board[i + p][j + q] - '0';
                        if (vis[v])
                            return false;

                        vis[v] = true;;
                    }
                }
            }
        }
        return true;
    }
};
```
