# [3446. 按对角线进行矩阵排序](https://leetcode.cn/problems/sort-matrix-by-diagonals/)

## 解题思路

1. 由于题目给定的输入数据范围比较小, ,按照题意模拟即可
1. 模拟过程可以分两段: 左下角和右上角


```cpp
using i32 = int;
using i64 = long long;
using pii = pair<int, int>;
using pll = pair<i64, i64>;
const int mod = 1000000007, mod0 = 998244353;
class Solution {
public:
    vector<vector<int>> sortMatrix(vector<vector<int>>& grid) {
        vector<vector<int>>& g = grid;
        int n = g.size();

        for (int i = 0; i < n; ++i) {
            int x = i, y = 0;
            vector<int> tmp;
            while (x < n && y < n) {
                tmp.push_back(g[x][y]);
                ++x;
                ++y;
            }
            sort(tmp.begin(), tmp.end(), greater<int>());

            x = i, y = 0;
            for (int &v : tmp) {
                g[x][y] = v;
                ++x;
                ++y;
            }
        }

        for (int j = 1; j < n; ++j) {
            int x = 0, y = j;
            vector<int> tmp;
            while (x < n && y < n) {
                tmp.push_back(g[x][y]);
                ++x;
                ++y;
            }
            sort(tmp.begin(), tmp.end());
            x = 0, y =j;
            for (int &v : tmp) {
                g[x][y] = v;
                ++x;
                ++y;
            }
        }
        return g;
    }
};
```
