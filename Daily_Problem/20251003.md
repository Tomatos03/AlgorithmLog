## 参考代码

+ 时间复杂度$O(n)$


```java
class Solution {
public:
    int trap(vector<int> &height) {
        int ans = 0, n = height.size();
        height.insert(height.begin(), INT_MAX);
        vector<int> dec;
        dec.push_back(0);
        for (int i = 1; i <= n; ++i) {
            while (height[i] > height[dec.back()]) {
                int h = min(height[i], height[dec[1]]) - height[dec.back()];
                int j = dec.back();
                dec.pop_back();
                int w = j - dec.back();
                ans += h * w;
            }
            dec.push_back(i);
        }
        return ans;
    }
};
```
